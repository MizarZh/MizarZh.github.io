Typescript基本使用

2022-01-29

类型

基础类型

// bool
let isDone: boolean = false;

// number
let decLiteral: number = 6;
let hexLiteral: number = 0xf00d;
let binaryLiteral: number = 0b1010;
let octalLiteral: number = 0o744;

// string
let name: string = "bob";

// array
let list: number[] = [1, 2, 3]; 
let list: Array<number> = [1, 2, 3]; // 数组泛型

// tuple
let x: [string, number];
x = ['hello', 10]; // OK
x = [10, 'hello']; // Error

// enum，从0开始编号，也可以用 Red = 1 指定
enum Color {Red, Green, Blue}
let c: Color = Color.Green;

// any，不规定任何值，和 Object 不同的是其成员也可以不确定。
let notSure: any = 4;
let list: any[] = [1, true, "free"];

// void 一个函数没有返回值时是void，只能被赋值 undefined 和 null
function warnUser(): void {
    alert("This is my warning message");
}

// null, undefined 没啥意思
let u: undefined = undefined;
let n: null = null; 

// never 永不存在的值的类型（不会到达终点）
function error(message: string): never {
    throw new Error(message);
}
function infiniteLoop(): never {
    while (true) {
    }
}

类型断言

相信我，我知道自己在干什么。（类型转换）

// <>
let someValue: any = "this is a string";
let strLength: number = (<string>someValue).length;

// as
let someValue: any = "this is a string";
let strLength: number = (someValue as string).length;

对象解构

接口

interface LabelledValue {
  label: string;
}

// labelledObj 必须符合 LabelledValue 接口的格式
function printLabel(labelledObj: LabelledValue) {
  console.log(labelledObj.label);
}

let myObj = {size: 10, label: "Size 10 Object"};
printLabel(myObj);

可选属性

有时候接口里的属性并不是全部都需要，这时候就需要可选属性（option bags）了：

interface SquareConfig {
  color?: string;
  width?: number;
}

只读属性

可以加入只读属性限定：

interface Point {
    readonly x: number;
    readonly y: number;
}

对于数组，可以使用 ReadonlyArray<T> 泛型。

对于变量，就是 const。

函数类型

interface SearchFunc {
  (source: string, subString: string): boolean;
}
let search: SearchFunc;
search = function(source: string, subString: string): boolean {
    ...// 具体实现
}

(..输入值类型..) : 返回值类型

可索引类型

可索引的类型（如数组 a[10] 和对象 people['name']）可以有一个索引签名。共有支持两种索引签名：字符串和数字。

interface StringArray {
  readonly [index: number]: string; // 可以设置 readonly，防止给索引赋值
}

let myArray: StringArray;
myArray = ["Bob", "Fred"];

let myStr: string = myArray[0];

额外属性检查

下面这段语句表示的是SquareConfig可以有任意数量的属性，并且只要它们不是color和width，那么就无所谓它们的类型是什么。

interface SquareConfig {
    color?: string;
    width?: number;
    [propName: string]: any;
}
